  • 常量
  // bad
  if (this.smsSendedCount > 1000) { // ??? 1000是什么
    console.log('短信发送数量大于1000')
  }
  // good
  const MAX_SMS_SENDED_COUNT = 1000
  if (this.smsSendedCount > MAX_SMS_SENDED_COUNT) {
    console.log('短信发送数量大于1000)
  }
  // bad
  const person = {} // 这样会把Object的方法都会继承过来
  // good
  const person = Object.create(null) // 会使得person比较纯粹
  • 参数默认值
  // bad
  function app(name) {
    name = name || '默认值'
  }
  // good
  function app(name = '默认值') {}
  // good 如果一定要传入参数，添加强提醒
  function isRequired() { throw new Error('params is required') }
  function app(name = isRequried()){}
  • 解构
  // bad
  const smsSendCounts = [1, 2]
  const gameSmsSendCount = smsSendCounts[0]
  const financeSmsSendCount = smsSendCounts[1]
  // good
  const [gameSmsSendCount, financeSmsSendCount] = smsSendCounts
  • 函数式风格
  // bad
  let sum = 0
  const smsSends = [1, 2, 3, 4]
  smsSends.forEach(item => {
    sum += item
  })
  // good
  const result = smsSends.reduce((prev, current) => {
    prev += current
    return prev
  }, 0)
  • 定义函数
  // bad
  function app() {} // 这种写法会被解析器自动提升到代码头部，违背了js先定义后使用的要求
  // good
  const app = function () {}

  const fetchXXX = () => {} // 调用get api

    const res = await fetchXXX()

  }
  // bad
  const branchList = []
  // good
  const branchs = [] // 禁止出现xxxList

  const publish = ({ param1, param2, param3, param4 }) => {} // 超过3个参数用对象的形式传参
  // bad

  const isSupport = () => {} // 禁止【 否定 】 定义函数名
