-  Promise

动手实现简易版的Promise后，[Promise是一个容器，包含着很多未来才会执行的事件]
```javascript
class MyPromise {
  constructor(callback) {
    // 处理返回成功的逻辑
    let value = null
    let isSuccess = null
    const arr = []
    this.then = (thenFn) => {
      if (isSuccess === true) {
        arr.push(thenFn(arr.length === 0 ? value : arr[arr.length - 1]))
      }
      return this
    }
    // 处理返回失败的逻辑
    this.catch = (catchFn) => {
      if (isSuccess === false) {
        catchFn(value)
      }
      return this
    }
    function resolve(data) {
      isSuccess = true
      value = data
    }
    function reject(data) {
      isSuccess = false
      value = data
    }
    callback(resolve, reject)
  }
}
```
